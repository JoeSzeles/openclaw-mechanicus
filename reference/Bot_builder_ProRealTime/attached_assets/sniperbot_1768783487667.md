//-------------------------------------------------------------------------
Defparam CumulateOrders = False

// --- Settings ---
PositionSize = 0.5
TargetPoints = 100
MaxDrawdownAUD = 1000

// --- Step 1: Entry Condition ---
// We use StrategyProfit to see if we have moved yet.
// If profit is 0 and we aren't in a trade, we FIRE.
IF StrategyProfit = 0 AND NOT ONMARKET THEN
BUY PositionSize CONTRACTS AT MARKET
SET TARGET pPROFIT TargetPoints
SET STOP $LOSS MaxDrawdownAUD
ENDIF

// --- Step 2: The Killswitch ---
// Once the position is closed, StrategyProfit will no longer be 0.
// This tells the bot the "Sniper Shot" is complete.
IF StrategyProfit <> 0 AND NOT ONMARKET THEN
QUIT
ENDIF
